\subsubsection{Sigma Delta Modulator}
\begin{minipage}[t]{0.3\textwidth}
	\vspace{0pt}
	\includegraphics[width=\textwidth]{"SigmaDeltaModulator"}
\end{minipage}\hspace{0.05\textwidth}
\begin{minipage}[t]{0.3\textwidth}
	\vspace{0pt}
	\formula{$0 \approx Q_{Cint} = \dfrac{N T_{clk} V_{in}}{R_{i1}} + \dfrac{n T_{clk} V_{refp}}{R_{i2}} + \dfrac{(N -n) T_{clk} V_{refp}}{R_{i2}}$}\\
	\formula{$n = N \cdot \dfrac{V_{in} + V_{ref}}{2 V_{ref}}$}\\
	\formula{$V_{in} \dfrac{2 n - N}{N} V_{ref}$}
\end{minipage}\hspace{0.05\textwidth}
\begin{minipage}[t]{0.3\textwidth}
	\vspace{0pt}
	2 Zähler, einer zählt die CLK-Perioden, der andere zählt die Perioden, in denen der Komparator = 1 ist.\\
	\unitText{$N$}{Anzahl Taktzyklen}\\
	\unitText{$n$}{Anzahl Taktperioden, bei denen denen der Modulatorausgang = 1}
\end{minipage}
